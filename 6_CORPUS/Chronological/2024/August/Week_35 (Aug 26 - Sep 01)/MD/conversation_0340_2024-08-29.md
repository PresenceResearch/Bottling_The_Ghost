from docx import Document

# Load the two documents
doc1_path = "/mnt/data/PHIL Discovery Call 1-Page Outline.docx"
doc2_path = "/mnt/data/SHANE Discovery Call Steps and Flow.docx"

doc1 = Document(doc1_path)
doc2 = Document(doc2_path)

# Create a new Document for the master document
master_doc = Document()

# Title for the master document
master_doc.add_heading("Synthesis of Discovery Call Guides", level=1)

# Function to add a paragraph with attribution
def add_paragraph_with_attribution(text, attribution):
    para = master_doc.add_paragraph()
    run = para.add_run(f"{text} ")
    attribution_run = para.add_run(f"[{attribution}]")
    attribution_run.italic = True
    attribution_run.bold = True

# Extract and synthesize sections
sections_doc1 = [para.text for para in doc1.paragraphs if para.text.strip() != ""]
sections_doc2 = [para.text for para in doc2.paragraphs if para.text.strip() != ""]

# Combined content with attribution
combined_sections = []
for section in sections_doc1 + sections_doc2:
    if section in sections_doc1 and section not in combined_sections:
        add_paragraph_with_attribution(section, "Phil")
        combined_sections.append(section)
    elif section in sections_doc2 and section not in combined_sections:
        add_paragraph_with_attribution(section, "Shane")
        combined_sections.append(section)

# Save the compiled master document
output_path = "/mnt/data/Synthesized_Discovery_Call_Guide.docx"
