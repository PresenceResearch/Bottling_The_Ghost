```python
import os, sys, json, csv, argparse
from datetime import datetime

def get_file_timestamp(path):
    try:
        ts = os.path.getmtime(path)
        return datetime.fromtimestamp(ts).isoformat()
    except Exception:
        return None

def normalize_row(filename, timestamp, actor, text):
    return {
        "filename": filename or "",
        "timestamp": (timestamp or "").strip() or get_file_timestamp(filename) or "",
        "actor": (actor or "").strip(),
        "text": (text if isinstance(text, str) else json.dumps(text, ensure_ascii=False)).strip(),
    }

def process_json_file(filepath, rows, log):
    try:
        with open(filepath, "r", encoding="utf-8", errors="ignore") as f:
            data = json.load(f)
    except Exception as e:
        log.append(f"[ERROR] Could not read JSON: {filepath} :: {e}")
        return

    # Case A: list of messages
    if isinstance(data, list):
        for entry in data:
            rows.append(normalize_row(
                os.path.basename(filepath),
                (entry.get("timestamp") or entry.get("time") or entry.get("created_at") or None),
                (entry.get("actor") or entry.get("sender") or entry.get("role") or ""),
                (entry.get("text") or entry.get("message") or entry.get("content") or entry),
            ))
        return

    # Case B: dict with "messages"
    if isinstance(data, dict) and "messages" in data and isinstance(data["messages"], list):
        for entry in data["messages"]:
            rows.append(normalize_row(
                os.path.basename(filepath),
                (entry.get("timestamp") or entry.get("time") or entry.get("created_at") or None),
                (entry.get("actor") or entry.get("sender") or entry.get("role") or ""),
                (entry.get("text") or entry.get("message") or entry.get("content") or entry),
            ))
        return

    # Fallback: dump whole object as one row
    rows.append(normalize_row(os.path.basename(filepath), get_file_timestamp(filepath), "", data))

def process_txt_file(filepath, rows, log):
    try:
        with open(filepath, "r", encoding="utf-8", errors="ignore") as f:
            content = f.read().strip()
        rows.append(normalize_row(os.path.basename(filepath), get_file_timestamp(filepath), "", content))
    except Exception as e:
        log.append(f"[ERROR] Could not read TXT: {filepath} :: {e}")

def walk_target(path):
    files = []
    if os.path.isfile(path):
        files = [path]
    else:
        for root, _, flist in os.walk(path):
            for name in flist:
                files.append(os.path.join(root, name))
    return files

def main():
    parser = argparse.ArgumentParser(description="Flatten Brandy TXT/JSON into CSV")
    parser.add_argument("--path", required=True, help="File or folder path")
    parser.add_argument("--out", default="brandy_archive.csv", help="Output CSV filename")
    args = parser.parse_args()

    target = args.path
    rows, log = [], []

    if not os.path.exists(target):
        print(f"[FATAL] Path not found: {target}")
        sys.exit(1)

    files = walk_target(target)
    if not files:
        print(f"[WARN] No files found under: {target}")
        sys.exit(0)

    found_any = False
    for fp in files:
        ext = os.path.splitext(fp)[1].lower()
        if ext == ".json":
            process_json_file(fp, rows, log); found_any = True
        elif ext == ".txt":
            process_txt_file(fp, rows, log); found_any = True
        else:
            # ignore other extensions silently, but you can log if you want
            pass

    # Write CSV
    if found_any and rows:
        with open(args.out, "w", newline="", encoding="utf-8") as csvfile:
            fieldnames = ["filename", "timestamp", "actor", "text"]
            writer = csv.DictWriter(csvfile, fieldnames=fieldnames)
            writer.writeheader()
            writer.writerows(rows)

        print(f"[OK] Extracted {len(rows)} rows into {args.out}")
    else:
        print("[WARN] No .txt or .json content extracted.")

    # Print log lines last
    if log:
        print("\n[LOG]")
        for line in log:
            print(line)

if __name__ == "__main__":
    main()
```

Yep — here’s a quick Python script you can run locally to split a **large `conversations.json`** into smaller JSON files, each under ~50 MB so they’re safe to upload here.  

---

### **split_json.py**
```python
import json
import os

INPUT_FILE = "conversations.json"   # your big JSON file
OUTPUT_DIR = "chunks"               # folder to save the chunks
CHUNK_SIZE = 50 * 1024 * 1024        # 50 MB in bytes

# Make output folder
os.makedirs(OUTPUT_DIR, exist_ok=True)

# Load the JSON (must be a list at top-level)
with open(INPUT_FILE, "r", encoding="utf-8") as f:
    data = json.load(f)

if not isinstance(data, list):
    raise ValueError("JSON root must be a list of messages/objects")

chunk = []
chunk_bytes = 0
chunk_num = 1

for entry in data:
    entry_str = json.dumps(entry, ensure_ascii=False) + "\n"
    size = len(entry_str.encode("utf-8"))

    if chunk_bytes + size > CHUNK_SIZE:
        out_path = os.path.join(OUTPUT_DIR, f"chunk_{chunk_num}.json")
        with open(out_path, "w", encoding="utf-8") as out_f:
            json.dump(chunk, out_f, ensure_ascii=False, indent=2)
        print(f"Saved {out_path} ({chunk_bytes/1024/1024:.2f} MB)")
        chunk_num += 1
        chunk = []
        chunk_bytes = 0

    chunk.append(entry)
    chunk_bytes += size

# Save the final chunk
if chunk:
    out_path = os.path.join(OUTPUT_DIR, f"chunk_{chunk_num}.json")
    with open(out_path, "w", encoding="utf-8") as out_f:
        json.dump(chunk, out_f, ensure_ascii=False, indent=2)
    print(f"Saved {out_path} ({chunk_bytes/1024/1024:.2f} MB)")

print("Done.")
```
