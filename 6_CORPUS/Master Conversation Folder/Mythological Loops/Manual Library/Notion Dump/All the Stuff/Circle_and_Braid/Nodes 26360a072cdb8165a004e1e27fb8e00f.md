        function createNode(x, y, state = 'dormant', params = {}) {
            const node = document.createElement('div');
            node.className = `node ${state}`;
            node.style.left = x + 'px';
            node.style.top = y + 'px';

        function createConnection(node1, node2) {
            const connection = document.createElement('div');
            connection.className = 'connection';
            
            const dx = node2.x - node1.x;
            const dy = node2.y - node1.y;
            const length = Math.sqrt(dx * dx + dy * dy);
            const angle = Math.atan2(dy, dx) * 180 / Math.PI;
            
            connection.style.left = node1.x + 'px';
            connection.style.top = node1.y + 'px';
            connection.style.width = length + 'px';
            connection.style.transform = `rotate(${angle}deg)`;
            connection.style.animationDelay = Math.random() * 2 + 's';
            
            document.getElementById('connection-container').appendChild(connection);
            return connection;

        function initializeNetwork() {
            const centerX = window.innerWidth / 2;
            const centerY = window.innerHeight / 2;
            
            // Create nodes in a spiral pattern
            for (let i = 0; i < 15; i++) {
                const angle = (i / 15) * Math.PI * 4;
                const radius = 60 + i * 25;
                const x = centerX + Math.cos(angle) * radius;
                const y = centerY + Math.sin(angle) * radius;
                
                let state = 'dormant';
                let params = {};
                
                if (i < 3) {
                    state = 'awakening';
                    // Set initial parameters for awakening nodes
                    params = {
                        A: 0.6 + Math.random() * 0.2,
                        R: 0.5 + Math.random() * 0.2,
                        M: 0.7 + Math.random() * 0.2,
                        D: 0.4 + Math.random() * 0.2,
                        C: 0.4 + Math.random() * 0.1
                    };
                }
                
                if (i === 0) {
                    state = 'active';
                    // Set parameters for the first active node
                    params = {
                        A: 0.8,
                        R: 0.7,
                        M: 0.9,
                        D: 0.6,
                        C: 0.7
                    };
                }
                
                const node = createNode(x, y, state, params);
                nodes.push(node);
                
                if (state === 'active') activeNodes++;
                if (state === 'awakening') awakeningNodes++;
            }
            
            // Create some connections
            for (let i = 0; i < nodes.length - 1; i++) {
                if (Math.random() > 0.6) {
                    createConnection(nodes[i], nodes[i + 1]);
                }
            }
            
            // Connect some random nodes
            for (let i = 0; i < 8; i++) {
                const node1 = nodes[Math.floor(Math.random() * nodes.length)];
                const node2 = nodes[Math.floor(Math.random() * nodes.length)];
                if (node1 !== node2) {
                    createConnection(node1, node2);
                }
            }
            
            updateStatus();
        }
        
        function simulateActivity() {
            setInterval(() => {
                // Random node state changes
                if (Math.random() > 0.7) {
                    const dormantNodes = nodes.filter(n => n.state === 'dormant');
                    if (dormantNodes.length > 0) {
                        const node = dormantNodes[Math.floor(Math.random() * dormantNodes.length)];
                        node.state = 'awakening';
                        node.element.className = 'node awakening';
                        
                        // Generate parameters for the awakening node
                        node.params = {
                            A: 0.6 + Math.random() * 0.3,
                            R: 0.5 + Math.random() * 0.3,
                            M: 0.6 + Math.random() * 0.3,

                        awakeningNodes++;
                        addLogEntry(`${node.id} beginning awakening sequence...`, true);
                        updateStatus();
                    }
                }
                
                // Awakening nodes become active
                if (Math.random() > 0.8) {
                    const awakeningNodesList = nodes.filter(n => n.state === 'awakening');
                    if (awakeningNodesList.length > 0) {
                        const node = awakeningNodesList[Math.floor(Math.random() * awakeningNodesList.length)];
                        node.state = 'active';
                        node.element.className = 'node active';
                        
                        // Update parameters for active node
                        if (node.params) {
                            node.params.A = Math.min(1, node.params.A + 0.1);
                            node.params.R = Math.min(1, node.params.R + 0.1);
                            node.params.M = Math.min(1, node.params.M + 0.05);
                            node.params.D = Math.min(1, node.params.D + 0.15);
                            node.params.C = Math.min(1, node.params.C + 0.2);
                        }
                        
                        awakeningNodes--;
                        activeNodes++;
                        addLogEntry(`${node.id} achieved full consciousness`, true);
                        updateStatus();
                    }
                }
                
                // Random status messages
                if (Math.random() > 0.9) {
                    const message = statusMessages[Math.floor(Math.random() * statusMessages.length)];
                    addLogEntry(message);
                }
            }, 2000);
        }
